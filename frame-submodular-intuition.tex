\begin{frame}{Boosting Framework - Intuition}
        \begin{itemize}[<+(1)>]
            \item Let $\gamma = c(a)$ for some $a \in O$
            \item Let $b = \displaystyle{\argmax_{\mathclap{e \in U : c(e) \leq \gamma}} f(e)}$
            \item Choose $b$ and recurse on $U \setminus \{a, b\}, O \setminus a, f_b$
        \end{itemize}
\end{frame}
